-- Countdown function (3x faster)
local function runCountdown(button, action)
    if button:GetAttribute("Running") then return end
    button:SetAttribute("Running", true)

    local originalText = button.Text
    for i = 4, 1, -1 do -- only 4 steps instead of 10
        button.Text = "<b>Wait "..i.."s</b>"
        task.wait(0.33) -- each "second" lasts ~0.33 real seconds
    end

    -- Execute the action after countdown
    action()

    button.Text = originalText
    button:SetAttribute("Running", false)
end


-- Dupe Equipped Item Logic
dupeBtn.MouseButton1Click:Connect(function()
    runCountdown(dupeBtn, function()
        local character = player.Character
        if character then
            local tool = character:FindFirstChildOfClass("Tool")
            if tool and not tostring(tool):match("Seed") then
                for i = 1, multiplier do
                    local cloneTool = tool:Clone()
                    cloneTool.Parent = player.Backpack
                end
            end
        end
    end)
end)

-- Dupe Seeds Logic
dupeSeedsBtn.MouseButton1Click:Connect(function()
    runCountdown(dupeSeedsBtn, function()
        local character = player.Character
        if character then
            local tool = character:FindFirstChildOfClass("Tool")
            if tool and tostring(tool):match("Seed") then
                tool.Name = tostring(tool):gsub(
                    tostring(tool):match("(%d+)"),
                    tonumber(tostring(tool):match("(%d+)") or 1) * multiplier
                )
            end
        end
    end)
end)
